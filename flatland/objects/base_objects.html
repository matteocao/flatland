 98    def update(self, event: Any) -> bool:
 99        if self.is_prepare_just_done:
100            self.is_prepare_just_done = False
101            self.logger.info(f"Update for {self.__class__.__name__}")
102            self.volition.update()
103            self.scheduler.update()  # this runs all teh interaction callables
104            # animation flags:
105            if self.prev_x != self.x or self.prev_y != self.y:
106                self.is_pushing = False
107                self.is_moving = True
108                self.is_standing = False
109            elif (
110                self.prev_x == self.x
111                and self.prev_y == self.y
112                and any("push" == func.__name__ for func, _ in self.volition.list_of_actions)
113            ):
114                self.is_pushing = True
115                self.is_moving = False
116                self.is_standing = False
117            elif self.prev_x == self.x and self.prev_y == self.y:
118                self.is_pushing = False
119                self.is_moving = False
120                self.is_standing = True
121            return True
122        return False

124    def prepare(self, near_objs: Any, game: "Game") -> bool:
125        now = pygame.time.get_ticks()  # in ms
126        interval = 1000 / self.actions_per_second
127        tick = int(now // interval)
128
129        if tick != self.last_tick:
130            self.prev_x = self.x
131            self.prev_y = self.y
132            self.last_tick = tick
133            self.logger.info(f"Preparation for {self.__class__.__name__}")
134            # prepare interactions
135            for near_obj in near_objs:
136                self.scheduler.add(InteractionCommand(self, near_obj, game))
137            self.volition.prepare(game)
138            self.internal_state.update(near_objs)
139            self.is_prepare_just_done = True
140            return True
141        return False

143    def distance(self, other: Any) -> float:
144        return math.sqrt((self.x - other.x) ** 2 + (self.y - other.y) ** 2)
