218    def __init__(self, x: int, y: int, name: str, health: int, **kwargs: Any):
219        super().__init__(x, y, name, health)
220        self.color = (0, 255, 0)
221        self.noise_intensity = 1.1

223        self.visible_size = 2.0
224
225    def get_pressed_keys(self, keys) -> None:
226        self.keys = keys
227
228    def prepare(self, near_objs: Any):
229        dx = dy = 0
230        if self.keys[pygame.K_UP]:
231            dy = -1
232        elif self.keys[pygame.K_DOWN]:
233            dy = 1
234        elif self.keys[pygame.K_LEFT]:
235            dx = -1
236        elif self.keys[pygame.K_RIGHT]:
237            dx = 1
238
239        self.new_x = (self.x + dx) % MAX_X
240        self.new_y = (self.y + dy) % MAX_Y
241
242    def update(self, event):
243        self.x = self.new_x
244        self.y = self.new_y
245
246    def render(self, screen):
247        pygame.draw.rect(

249            self.color,
